        <div>

            <p id="total-expense" class="text-lg font-semibold text-gray-700">₺0</p>
        </div>

    <div class="card mb-6">
        <h3 class="font-bold mb-3 text-gray-700">Yeni İşlem</h3>
        <input type="text" id="desc" placeholder="Açıklama (örn: Market)" class="w-full border p-2 rounded mb-2 outline-none focus:ring-2 focus:ring-indigo-300">
        <input type="number" id="amount" placeholder="Miktar (₺)" class="w-full border p-2 rounded mb-3 outline-none focus:ring-2 focus:ring-indigo-300">
        <div class="flex gap-2">
            <button onclick="addTransaction('income')" class="flex-1 bg-green-500 text-white py-2 rounded-lg active:scale-95 transition">Gelir</button>
            <button onclick="addTransaction('expense')" class="flex-1 bg-red-500 text-white py-2 rounded-lg active:scale-95 transition">Gider</button>
        </div>
    </div>

    <div class="card">
        <h3 class="font-bold mb-3 text-gray-700">Son İşlemler</h3>
        <ul id="list" class="divide-y">
            </ul>
    </div>

    <script>
        let transactions = JSON.parse(localStorage.getItem('my_budget')) || [];

        function addTransaction(type) {
            const desc = document.getElementById('desc').value;
            const amount = document.getElementById('amount').value;

            if (desc.trim() === '' || amount === '') return;

            const transaction = {
                id: Date.now(),
                desc,
                amount: type === 'expense' ? -Math.abs(amount) : Math.abs(amount)
            };

            transactions.push(transaction);
            updateUI();
            saveData();
            
            document.getElementById('desc').value = '';
            document.getElementById('amount').value = '';
        }

        function updateUI() {
            const list = document.getElementById('list');
            const totalBalance = document.getElementById('total-balance');
            const totalIncome = document.getElementById('total-income');
            const totalExpense = document.getElementById('total-expense');

            list.innerHTML = '';
            let income = 0;
            let expense = 0;

            transactions.slice().reverse().forEach(t => {
                const item = document.createElement('li');
                item.className = 'flex justify-between py-3 text-sm';
                item.innerHTML = `
                    <span>${t.desc}</span>
                    <span class="${t.amount > 0 ? 'text-green-600' : 'text-red-600'} font-bold">
                        ${t.amount > 0 ? '+' : ''}${t.amount} ₺
                    </span>
                `;
                list.appendChild(item);

                if (t.amount > 0) income += t.amount;
                else expense += t.amount;
            });

            const total = income + expense;
            totalBalance.innerText = `₺${total.toFixed(2)}`;
            totalIncome.innerText = `₺${income}`;
            totalExpense.innerText = `₺${Math.abs(expense)}`;
        }

        function saveData() {
            localStorage.setItem('my_budget', JSON.stringify(transactions));
        }

        updateUI();
    </script>
